Оператор приращения
a = 2

Примеры:
+=   #увеличение переменной сложением
-=   #уменьшение переменой вычитанием
*=   #умножение
/=   #вещественной деление
//=  #целочисленное деление
%=   #остаток
**=  #возведение в степень
